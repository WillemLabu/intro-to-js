Event handling


  <div class="container">
    <button>Show</button>
  </div>

  <div id="modal-box" style="display: none">
    I'm a modal, perhaps..
  </div>



  // Again, this targets 'button' elements within '.container'
  $('.container button').on('click', function() {
    //           _______________________^
    //          |
    // This function is run "on the click event"
    //  of all the matching ".container button" elements

    $('#modal-box').show();

  });



  <div class="container">
    <button>Show</button>
  </div>

  <div id="modal-box" class="hidden">
    I'm a modal, perhaps..
  </div>



  $('.container button').on('click', function() {

    $('#modal-box').removeClass('hidden');

  });
